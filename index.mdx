<Columns cols={2}>
  <Card title="Customer payouts" icon="bank">
    Initiate one-off or recurring payouts to marketplace sellers, suppliers, or gig workers.
  </Card>
  <Card title="Treasury automation" icon="arrows-rotate">

## Get started quickly

<Steps>
  <Step title="Request sandbox access" icon="key" href="mailto:support@useyala.com">
    Reach out to Yala Developer Support for sandbox credentials and your dedicated API key.
  </Step>
  <Step title="Quote payouts" icon="calculator" href="/api-reference/endpoint/calculate-exchange-rate">
    Use the exchange-rate calculator endpoint to determine fees, settlement timings, and FX rates.
  </Step>
  <Step title="Initiate transactions" icon="paper-plane" href="/api-reference/endpoint/initiate">
    Create payouts programmatically, attach compliance metadata, and track status changes via webhooks.
  </Step>
</Steps>
